用于构建用户界面的CoffeeScript库

柠檬是建立用户界面的极其快速的方式。 只需八千字节就可以让您获得反应式，可路由， 有状态的组件和适用于搜索引擎优化的应用程序，而无需使用JavaScript或HTML行。

轻量级和高性能

你是否曾经有过一个React.js项目， 其中你的node_modules目录增长超过一个千兆字节， 或者你的bundle.js达到了一兆字节？ 你不是一个人。 一个基本的柠檬应用程序可以建立零本地依赖和一个轻量级的命令行界面， 注入八千字节的JavaScript来运行你的应用程序。

服务器端渲染和搜索引擎优化

当您构建单页应用程序时，您会失去搜索引擎 优化。 当您构建服务器端呈现的动态网站时， 你失去了有状态的，被动的，基于组件的用户界面。 柠檬 给你们两个。 柠檬渲染你的应用服务器端，发送搜索引擎 优化了浏览器的HTML。 一旦进入浏览器，所有必要的 反应性数据侦听器和事件处理程序已附加。

灵活，直观的组件管理

组件应该易于管理。 你应该被允许选择 静态和动态数据之间。 你不应该同时管理两者 道具和状态。 您不必手动绑定每个事件 监听器。 你不应该清理超时和子组件。 柠檬使组件管理灵活，简单，直观。

事件处理程序

如何将常见事件侦听器添加到DOM

module.exports = lemon.Component {
  package: 'site'
  name: 'EventHandlers'

  data: {
    output: '[event data will show up here]'
  }

  methods: {
    handleEvent: (event) ->
      @output = "[#{event.type}] on #{event.target.innerText}"
  }

  template: (data) ->

    # "on" property (nice when you need lots of handlers)
    button on: {
      mouseover: "handleEvent"
      mouseout: "handleEvent"
      click: "handleEvent"
    }, ->
      "Mouseover Me"

    # "$" shorthand (convenient when you need one handler)
    button $click: 'handleEvent', ->
      "Click Me"

    # div for test output
    div _text: 'output'

}

数据绑定

如何动态使用数据进行部分更新

module.exports = lemon.Component {
  package: 'site'
  name: 'DataBinding'

  data: {
    output: 0
  }

  methods: {
    onClick: (event) ->
      @output = Math.random()
  }

  template: (data) ->
    button $click: 'onClick', ->
      "Click Me"

    div bind: {text: 'output'}
    div _text: 'output'
    div _on: 'output', _template: (output) ->
      span -> "[#{output}]"

}

漂亮的if语句和循环渲染逻辑

module.exports = lemon.Component {
  package: 'site'
  name: 'StaticFlowControl'

  data: {
    items: ['a', 'b']
    visible: true
  }

  methods: {
    update: ->
      @visible = not @visible
      @mount()
  }

  template: (data) ->
    {items, visible} = data

    button $click: 'update', ->
      "Click to toggle visible"

    div ->
      "This element will be redrawn: #{Math.random()}"

    if visible
      for item in items
        div -> item

}

使用部分模板更新进行条件渲染

module.exports = lemon.Component {
  package: 'site'
  name: 'DynamicFlowControl'

  data: {
    items: ['a', 'b']
    visible: true
  }

  methods: {
    update: ->
      @visible = not @visible
  }

  template: ->

    button $click: 'update', ->
      "Click to toggle visible"

    div ->
      "This element will NOT be redrawn: #{Math.random()}"

    div _on: 'visible', _template: (visible, data) ->
      if visible
        for item in data.items or []
          div -> item

Dynamic Lists

高效处理DOM中的大型列表

module.exports = lemon.Component {
  package: 'site'
  name: 'DynamicLists'

  data: {
    counter: 0
    items: []
  }

  methods: {
    onPush: ->    @items.push(++@counter)
    onPop: ->     @items.pop()
    onShift: ->   @items.shift()
    onUnshift: -> @items.unshift(++@counter)
    onReset: ->   @items = []
    onReverse: -> @items.reverse()
    onSort: ->    @items.sort()
  }

  template: (data) ->

    for action in ['Push', 'Pop', 'Unshift', 'Shift', 'Reset', 'Reverse', 'Sort']
      button $click: "on#{action}", ->
        action

    div _list: 'items', _template: (item) ->
      span ->
        "#{item}"
}

Lifecycle Hooks

在组件寿命的不同阶段后采取行动

module.exports = lemon.Component {
  package: 'site'
  name: 'LifecycleHooks'

  data: {
    events: []
  }

  lifecycle: {
    beforeCreate: ->  # @events not available yet
    created: ->   @events.push 'created'
    beforeMount: ->  @events.push 'beforeMount'
    mounted: ->   @events.push 'mounted'
    beforeDestroy: -> @events.push 'beforeDestroy'
    destroyed: ->  @events.push 'destroyed'
  }

  template: (data) ->

    div _list: 'events', _template: (event) ->
      span -> event

    button $click: '_destroy', ->
      'Click to destroy this component'
}
